⚙️ ตั้งค่า Google Sheets Sync

1
เปิด Google Sheets ใหม่

ไปที่ sheets.new → สร้าง Spreadsheet ใหม่ ตั้งชื่อว่า "Travel Tracker"

2
เปิด Apps Script

ใน Google Sheets → เมนู Extensions → Apps Script

3
วางโค้ดนี้แทนที่ทั้งหมด แล้วกด Save (💾)
const SHEET_NAME = 'expenses'; function doGet(e) { const ss = SpreadsheetApp.getActiveSpreadsheet(); let sheet = ss.getSheetByName(SHEET_NAME); if(!sheet) sheet = ss.insertSheet(SHEET_NAME); const data = sheet.getDataRange().getValues(); if(data.length <= 1) return respond({expenses:[],meta:{}}); const [headers,...rows] = data; const expenses = rows.map(r => rowToObj(headers,r)); const metaSheet = ss.getSheetByName('meta'); const meta = metaSheet ? JSON.parse(metaSheet.getRange('A1').getValue()||'{}') : {}; return respond({expenses,meta}); } function doPost(e) { const body = JSON.parse(e.postData.contents); const ss = SpreadsheetApp.getActiveSpreadsheet(); if(body.action === 'save_all') { let sheet = ss.getSheetByName(SHEET_NAME); if(!sheet) sheet = ss.insertSheet(SHEET_NAME); sheet.clearContents(); const headers = ['id','description','amount','category','paidBy','date','note']; const rows = body.expenses.map(ex => headers.map(h => ex[h]||'')); if(rows.length > 0) sheet.getRange(1,1,1,headers.length).setValues([headers]); if(rows.length > 0) sheet.getRange(2,1,rows.length,headers.length).setValues(rows); let ms = ss.getSheetByName('meta'); if(!ms) ms = ss.insertSheet('meta'); ms.getRange('A1').setValue(JSON.stringify(body.meta||{})); return respond({ok:true}); } return respond({ok:false}); } function rowToObj(headers,row){const o={};headers.forEach((h,i)=>o[h]=row[i]);o.amount=parseFloat(o.amount)||0;o.id=o.id||Date.now();return o;} function respond(data){return ContentService.createTextOutput(JSON.stringify(data)).setMimeType(ContentService.MimeType.JSON);}
4
Deploy เป็น Web App

กด Deploy → New deployment → เลือก Web app
ตั้งค่า "Execute as: Me" และ "Who has access: Anyone" → กด Deploy → อนุญาต (Allow)

💡 จะได้ URL ยาวๆ ที่ขึ้นต้นด้วย https://script.google.com/macros/s/...
